在父工程pom.xml的<project>元素中加入
  <build>
    	<resources> 
	  	<resource> 
	  	<directory>src/main/java</directory> 
	  	<includes> 
	  	<include>**/*.xml</include> 
	  	</includes> 
	  	<filtering>false</filtering>
		</resource> 
		<resource>
		<directory>src/main/resources</directory>
		</resource>
	</resources>
 </build>
